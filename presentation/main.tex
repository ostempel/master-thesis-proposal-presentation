\section{Introduction}
\begin{frame}{Introduction}

\begin{itemize}
  \item Kubernetes and cloud-native technologies have become industry standard
  \item Key challenges: perceived complexity, lack of training \cite{cncf2024}
  \item Root cause analysis (RCA) as key bottleneck
\end{itemize}

\vspace{10pt}

$\Rightarrow$\ LLM-based assistance systems promise cognitive reduction and timely detection

\end{frame}

\begin{itemize}

\begin{frame}[fragile]{Background}

\begin{columns}
\column{0.4\textwidth}
\textbf{Synergy-RCA} \cite{synergy-rca}:
\begin{itemize}
    \item periodically maps cluster snapshots to graph
    \item leverages LLMs to generate queries, checks, reports
\end{itemize}

\vspace{10pt}

\textbf{Limitations:}
\begin{itemize}
    \item not live $\rightarrow$ doesn't capture all events
    \item heavylifting of LLM
\end{itemize}
